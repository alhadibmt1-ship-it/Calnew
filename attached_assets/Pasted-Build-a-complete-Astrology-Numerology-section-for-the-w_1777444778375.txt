// npm install axios cheerio

const axios = require('axios');
const cheerio = require('cheerio');

const BASE = 'https://calcsmart24.com';
const PAGES = [
  '/astrology-numerology',
  '/astrology-numerology/birth-chart-calculator',
  '/astrology-numerology/synastry-chart-calculator',
  '/astrology-numerology/love-compatibility-calculator',
  '/astrology-numerology/sun-sign-calculator',
  '/astrology-numerology/moon-sign-calculator',
  '/astrology-numerology/rising-sign-calculator',
  '/astrology-numerology/big-three-calculator',
  '/astrology-numerology/venus-sign-calculator',
  '/astrology-numerology/daily-horoscopes',
  '/astrology-numerology/north-node-calculator',
  '/astrology-numerology/chiron-sign-calculator',
  '/astrology-numerology/part-of-fortune-calculator',
  '/astrology-numerology/lilith-vertex-calculator',
];

async function audit(path) {
  const url = BASE + path;
  const { data } = await axios.get(url, { timeout: 10000 });
  const $ = cheerio.load(data);
  const title = $('title').text().trim();
  const desc = $('meta[name="description"]').attr('content') || '';
  const h1s = $('h1').map((_, el) => $(el).text().trim()).get();
  const canonical = $('link[rel="canonical"]').attr('href') || 'MISSING';
  const viewport = $('meta[name="viewport"]').attr('content') || 'MISSING';
  const schemaCount = $('script[type="application/ld+json"]').length;
  const imgNoAlt = $('img:not([alt])').length;
  const ogTitle = $('meta[property="og:title"]').attr('content') || 'MISSING';

  const issues = [];
  if (title.length > 60) issues.push('Title too long: ' + title.length + ' chars');
  if (!title) issues.push('Title tag missing');
  if (desc.length < 120 || desc.length > 165) issues.push('Meta desc length off: ' + desc.length);
  if (h1s.length !== 1) issues.push('H1 count should be 1, got: ' + h1s.length);
  if (canonical === 'MISSING') issues.push('Canonical tag missing');
  if (viewport === 'MISSING') issues.push('Viewport meta missing');
  if (schemaCount === 0) issues.push('No JSON-LD schema found');
  if (imgNoAlt > 0) issues.push(imgNoAlt + ' image(s) missing alt text');
  if (ogTitle === 'MISSING') issues.push('og:title missing');

  return {
    page: path,
    title: title.slice(0, 55) + (title.length > 55 ? '…' : ''),
    h1: h1s[0] ? h1s[0].slice(0, 50) : 'MISSING',
    schema: schemaCount,
    issues: issues.length ? issues.join(' | ') : 'PASS'
  };
}

(async () => {
  console.log('\nCalcSmart24 — Astrology & Numerology Audit\n');
  for (const p of PAGES) {
    try {
      const r = await audit(p);
      console.log('PAGE:', r.page);
      console.log('  Title:   ', r.title);
      console.log('  H1:      ', r.h1);
      console.log('  Schema:  ', r.schema, 'JSON-LD block(s)');
      console.log('  Status:  ', r.issues);
      console.log('');
    } catch (e) {
      console.log('PAGE:', p, '— ERROR:', e.message, '\n');
    }
  }
})();

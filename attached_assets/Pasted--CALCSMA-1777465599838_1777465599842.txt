function wordCount(text) {
  return text.trim().split(/\s+/).filter(w => w.length > 1).length;
}

function severity(issues) {
  const score = issues.reduce((acc, i) => {
    if (i.includes('[CRITICAL]')) return acc - 20;
    if (i.includes('[HIGH]')) return acc - 10;
    if (i.includes('[MEDIUM]')) return acc - 5;
    if (i.includes('[LOW]')) return acc - 2;
    return acc;
  }, 100);
  return Math.max(0, score);
}

async function auditPage(path) {
  const url = BASE + path;
  let html, status;
  try {
    const res = await axios.get(url, {
      timeout: 15000,
      validateStatus: s => true,
      headers: { 'User-Agent': 'CalcSmart24-Auditor/1.0' }
    });
    status = res.status;
    html = res.data;
  } catch (e) {
    return { url, status: 'TIMEOUT/ERROR', issues: [`[CRITICAL] Page failed to load: ${e.message}`], score: 0 };
  }

  if (status !== 200) {
    return { url, status, issues: [`[CRITICAL] HTTP ${status} — page not returning 200`], score: 0 };
  }

  const $ = cheerio.load(html);
  const issues = [];

  // ── TITLE ──
  const titleEls = $('title');
  const title = titleEls.first().text().trim();
  if (titleEls.length === 0) issues.push('[CRITICAL] Missing <title> tag');
  else if (titleEls.length > 1) issues.push('[HIGH] Multiple <title> tags found');
  if (!title) issues.push('[CRITICAL] Empty title tag');
  else {
    if (title.length > 60) issues.push(`[HIGH] Title too long: ${title.length} chars (max 60)`);
    if (title.length < 30) issues.push(`[MEDIUM] Title too short: ${title.length} chars (min 30)`);
    if (titlesMap[title]) issues.push(`[CRITICAL] DUPLICATE TITLE — same as: ${titlesMap[title]}`);
    else titlesMap[title] = url;
  }

  // ── META DESCRIPTION ──
  const desc = $('meta[name="description"]').attr('content') || '';
  if (!desc) issues.push('[CRITICAL] Missing meta description');
  else {
    if (desc.length > 165) issues.push(`[HIGH] Meta desc too long: ${desc.length} chars`);
    if (desc.length < 120) issues.push(`[MEDIUM] Meta desc too short: ${desc.length} chars`);
    if (descsMap[desc]) issues.push(`[HIGH] DUPLICATE META DESCRIPTION — same as: ${descsMap[desc]}`);
    else descsMap[desc] = url;
  }

  // ── H1 ──
  const h1s = $('h1').map((_, el) => $(el).text().trim()).get();
  if (h1s.length === 0) issues.push('[CRITICAL] No H1 tag found');
  if (h1s.length > 1) issues.push(`[HIGH] Multiple H1 tags: ${h1s.length} found — only 1 allowed`);
  if (h1s[0]) {
    if (h1sMap[h1s[0]]) issues.push(`[HIGH] DUPLICATE H1 — same as: ${h1sMap[h1s[0]]}`);
    else h1sMap[h1s[0]] = url;
  }

  // ── HEADING HIERARCHY ──
  const headings = [];
  $('h1,h2,h3,h4,h5,h6').each((_, el) => headings.push(parseInt(el.tagName[1])));
  for (let i = 1; i < headings.length; i++) {
    if (headings[i] - headings[i-1] > 1) {
      issues.push(`[MEDIUM] Heading level skip: h${headings[i-1]} → h${headings[i]}`);
      break;
    }
  }

  // ── CANONICAL ──
  const canonical = $('link[rel="canonical"]').attr('href');
  if (!canonical) issues.push('[HIGH] Missing canonical tag');
  else if (canonical !== url && canonical !== url + '/')
    issues.push(`[HIGH] Canonical mismatch: got "${canonical}", expected "${url}"`);

  // ── VIEWPORT ──
  if (!$('meta[name="viewport"]').attr('content'))
    issues.push('[HIGH] Missing viewport meta tag — mobile indexing will fail');

  // ── LANG ──
  if (!$('html').attr('lang'))
    issues.push('[HIGH] Missing lang attribute on <html lang="en">');

  // ── SCHEMA ──
  const schemaCount = $('script[type="application/ld+json"]').length;
  if (schemaCount === 0) issues.push('[HIGH] No JSON-LD schema markup found');
  if (schemaCount === 1 && path.includes('/calculators/'))
    issues.push('[MEDIUM] Tool page should have 3 schema blocks: SoftwareApplication + FAQPage + BreadcrumbList');

  // ── IMAGES ──
  let imgNoAlt = 0, imgNoDims = 0, imgNoWebP = 0;
  $('img').each((_, el) => {
    const src = $(el).attr('src') || '';
    if (!$(el).attr('alt') && $(el).attr('alt') !== '') imgNoAlt++;
    if (!$(el).attr('width') || !$(el).attr('height')) imgNoDims++;
    if (src && !src.endsWith('.webp') && !src.includes('data:')) imgNoWebP++;
  });
  if (imgNoAlt > 0) issues.push(`[HIGH] ${imgNoAlt} image(s) missing alt attribute`);
  if (imgNoDims > 0) issues.push(`[MEDIUM] ${imgNoDims} image(s) missing width/height (causes CLS)`);
  if (imgNoWebP > 0) issues.push(`[LOW] ${imgNoWebP} image(s) not in WebP format`);

  // ── OPEN GRAPH ──
  ['og:title','og:description','og:image','og:url'].forEach(prop => {
    if (!$(`meta[property="${prop}"]`).attr('content'))
      issues.push(`[MEDIUM] Missing <meta property="${prop}">`);
  });

  // ── TWITTER CARD ──
  if (!$('meta[name="twitter:card"]').attr('content'))
    issues.push('[LOW] Missing twitter:card meta tag');

  // ── SEMANTIC STRUCTURE ──
  if ($('main').length === 0) issues.push('[HIGH] Missing <main> landmark element');
  if ($('header').length === 0) issues.push('[MEDIUM] Missing <header> element');
  if ($('footer').length === 0) issues.push('[MEDIUM] Missing <footer> element');
  if ($('nav').length === 0) issues.push('[MEDIUM] Missing <nav> element');

  // ── SKIP LINK ──
  const firstLink = $('a').first().attr('href');
  if (!firstLink || !firstLink.includes('#')) 
    issues.push('[MEDIUM] Missing skip-to-main-content link as first focusable element');

  // ── FORM ACCESSIBILITY ──
  let unlabelled = 0;
  $('input:not([type="hidden"]):not([type="submit"]), select, textarea').each((_, el) => {
    const id = $(el).attr('id');
    const ariaLabel = $(el).attr('aria-label');
    const ariaLabelledBy = $(el).attr('aria-labelledby');
    if (!ariaLabel && !ariaLabelledBy && (!id || $(`label[for="${id}"]`).length === 0))
      unlabelled++;
  });
  if (unlabelled > 0) issues.push(`[HIGH] ${unlabelled} form input(s) without accessible label`);

  // ── WORD COUNT ──
  const bodyText = $('body').clone().find('nav,header,footer,script,style').remove().end().text();
  const words = wordCount(bodyText);
  if (words < 300) issues.push(`[HIGH] Thin content: only ${words} words (minimum 300)`);
  else if (words < 400 && path.includes('/calculators/'))
    issues.push(`[MEDIUM] Tool page has ${words} words — recommend 400+ for SEO`);

  // ── DUPLICATE INTERNAL SECTIONS ──
  const sectionHeadings = $('h2, h3').map((_, el) => $(el).text().trim().toLowerCase()).get();
  const dupHeadings = sectionHeadings.filter((h, i) => sectionHeadings.indexOf(h) !== i);
  if (dupHeadings.length > 0)
    issues.push(`[MEDIUM] Duplicate section headings found: ${dupHeadings.slice(0,3).join(', ')}`);

  // ── DUPLICATE LINK DESTINATIONS ──
  const hrefs = $('main a[href]').map((_, el) => $(el).attr('href')).get();
  const hrefCounts = {};
  hrefs.forEach(h => { hrefCounts[h] = (hrefCounts[h] || 0) + 1; });
  const overlinked = Object.entries(hrefCounts).filter(([_, c]) => c > 2);
  if (overlinked.length > 0)
    issues.push(`[MEDIUM] These URLs linked 3+ times on same page: ${overlinked.map(([u]) => u).slice(0,2).join(', ')}`);

  // ── INLINE SCRIPTS WITH console.log ──
  let consoleLogs = 0;
  $('script:not([src])').each((_, el) => {
    if ($(el).html() && $(el).html().includes('console.log')) consoleLogs++;
  });
  if (consoleLogs > 0) issues.push(`[LOW] console.log() found in ${consoleLogs} inline script(s)`);

  return {
    url,
    status,
    title: title ? (title.length > 58 ? title.slice(0,55) + '…' : title) : 'MISSING',
    h1: h1s[0] ? (h1s[0].length > 50 ? h1s[0].slice(0,47) + '…' : h1s[0]) : 'MISSING',
    words,
    schemas: schemaCount,
    issues,
    score: severity(issues)
  };
}
